import './ExpenseItem.css';

const ExpenseItem = (props) => {

    const month = props.date.toLocaleString('en-US', { month: 'long'});
    const day = props.date.toLocaleString('en-US', { day: '2-digit'});

        <ExpenseDate date={props.date}/>
